pdu1

set ipaddress 192.168.1.7
set subnet 255.255.255.0
set gateway 192.168.1.1
set ipmode static
set dns1 8.8.8.8
set dns2 8.8.4.4
get network

set device name pdu1
set device temperature C
set device simplepassword true
get device

set time timezone GMT
get time

add user colibri
set user colibri password colibri
set user colibri role admin
set user colibri device local outlet all yes
get user colibri

set outlet 1 name firewall
set outlet 2 name access
set outlet 3 name services
set outlet 4 name control
set outlet 5 name detectors
set outlet 6 name blue
set outlet 7 name sparelinux
set outlet 8 name sparewindows
get outlets
